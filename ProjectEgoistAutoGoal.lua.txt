-- Função de Auto Goal (genérica, pode precisar de ajustes conforme o jogo)
local function getBall()
    for _, obj in ipairs(workspace:GetChildren()) do
        if obj.Name:lower():find("ball") then
            return obj
        end
    end
    return nil
end

local function getGoalPosition()
    -- Exemplo: encontra o gol mais próximo, pode ser necessário ajustar o nome do gol
    local minDist = math.huge
    local goalPos = nil
    for _, obj in ipairs(workspace:GetChildren()) do
        if obj.Name:lower():find("goal") then
            local dist = (obj.Position - game.Players.LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
            if dist < minDist then
                minDist = dist
                goalPos = obj.Position
            end
        end
    end
    return goalPos
end
